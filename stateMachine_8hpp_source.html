672  // It breaks, if there are no usable timeouts anymore
673  // It will (usually) not run until timeoutsQ is empty
674  while (!timeoutsQ.empty()) {
675  // Get the next timeout
676  auto timeoutElem = timeoutsQ.top();
677 
678  // If the current timeout is in the future -> break
679  if (timeoutElem.time > std::chrono::steady_clock::now()) {
680  break;

682

684  uint32_t timeoutID = timeoutElem.timeoutID;
685  auto timeoutDataIt = timeoutFunctions.find(timeoutID);
686

688  if (timeoutDataIt == timeoutFunctions.end()) {
689  timeoutsQ.pop();
690  continue;

692 
693  // Prepare function call
694  std::unique_ptr<struct TimeoutData> timeoutData =
695  std::move(timeoutDataIt->second);
696  auto stateIt = findState(timeoutData->id);
697  assert(stateIt != stateTable.end());
698  FunIface funIface(this, std::numeric_limits<uint32_t>::max(), pktsIn,
699  timeoutData->id, stateIt->second);
700 
701  // Clear the timeoutID from the state
702  stateIt->second.timeoutID = timeoutIDInvalid;
703 
704  // Call function
705  timeoutData->fun(stateIt->second, funIface);
706 
707  // Check if we reached the end state
708  if (stateIt->second.state == endStateID) {
709  D(std::cout << "Reached endStateID - deleting connection" << std::endl;)
710  removeState(timeoutData->id);
711  }
712 
713  // Clean up
714  timeoutsQ.pop();
715  timeoutFunctions.erase(timeoutDataIt);
716  }
717

719  for (uint32_t i = 0; i < inCount; i++) {
720  runPkt(pktsIn, i);
